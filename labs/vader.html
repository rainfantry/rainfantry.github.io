// LESSON 01 — DISCORD BOT SETUP

Intents, Bot, and the Message Loop

Build the foundation: Discord gateway connection, privileged intents, and the on_message handler that routes incoming messages to your LLM backend.

// CONCEPTS

  • Intents — Discord's permission system for what events your bot receives. message_content is privileged — must be enabled in the Discord Developer Portal.
  • commands.Bot — subclass of discord.Client that adds the command prefix system and extension loader.
  • on_message — raw event hook. Fires on EVERY message. You filter by trigger name, role, or channel before forwarding to the LLM.
  • asyncio event loop — Discord.py is fully async. Your LLM calls must be await-able or run in a thread executor.
  • dotenv — store the bot token in .env, never in source. load_dotenv(override=True) means .env wins over shell env vars.
// REFERENCE — mrrobot.py, intents + bot init
import discord
from discord.ext import commands
from dotenv import load_dotenv

load_dotenv(override=True)

BOT_TOKEN = os.getenv("DISCORD_BOT_TOKEN")
BOT_TRIGGER_NAMES = ["robot", "mrrobot", "mr robot"]

intents = discord.Intents.default()
intents.message_content = True        # privileged — enable in dev portal
intents.members = True

bot = commands.Bot(command_prefix="!", intents=intents)

@bot.event
async def on_ready():
    print(f"[+] logged in as {bot.user}")

@bot.event
async def on_message(message):
    if message.author == bot.user:
        return
    content = message.content.lower()
    triggered = any(name in content for name in BOT_TRIGGER_NAMES)
    if not triggered:
        await bot.process_commands(message)
        return
    # hand off to LLM handler
    await handle_llm(message)
    await bot.process_commands(message)

bot.run(BOT_TOKEN)

// LESSON 02 — PERSONA INJECTION

System Prompt as Personality Engine

The bot becomes whoever you tell it to be via a SYSTEM prompt prepended to every conversation. This is the core mechanism: inject a persona, maintain per-channel message history, forward to Ollama.

// CONCEPTS

  • SYSTEM_PROMPT — loaded from .env, defines the bot's personality, doctrine, constraints, and voice. The whole character lives here.
  • Per-channel history — collections.deque(maxlen=N) keeps the last N messages per channel ID. Sliding window keeps RAM bounded.
  • build_messages() — prepends the system prompt, appends history, then the current user message. Outputs the array Ollama's API expects.
  • Ollama /api/chat — POST to http://localhost:11434/api/chat with {"model": name, "messages": [...], "stream": false}. Returns response.message.content.
  • aiohttp — async HTTP client for non-blocking Ollama calls inside the Discord event loop.
// REFERENCE — persona injection + Ollama call
import os
import aiohttp
from collections import defaultdict, deque

SYSTEM_PROMPT = os.getenv("SYSTEM_PROMPT",
    "You are VADER, a machine-spirit serving the 22nd Survey Division. "
    "Respond with military precision. No corporate fluff.")

OLLAMA_URL  = os.getenv("OLLAMA_URL", "http://localhost:11434/api/chat")
MODEL_NAME  = os.getenv("MODEL_NAME", "huihui_ai/qwen2.5-coder-abliterate:7b")
HISTORY_DEPTH = int(os.getenv("HISTORY_DEPTH", "12"))

# per-channel sliding window
_history = defaultdict(lambda: deque(maxlen=HISTORY_DEPTH))

def build_messages(channel_id, user_content):
    msgs = [{"role": "system", "content": SYSTEM_PROMPT}]
    msgs.extend(_history[channel_id])
    msgs.append({"role": "user", "content": user_content})
    return msgs

async def query_ollama(channel_id, user_content):
    payload = {
        "model": MODEL_NAME,
        "messages": build_messages(channel_id, user_content),
        "stream": False,
    }
    async with aiohttp.ClientSession() as session:
        async with session.post(OLLAMA_URL, json=payload) as resp:
            data = await resp.json()
    reply = data["message"]["content"]
    _history[channel_id].append({"role": "user",    "content": user_content})
    _history[channel_id].append({"role": "assistant","content": reply})
    return reply

// LESSON 03 — VOICE CLONE + TTS

ElevenLabs PCM → WAV → Discord

Feed the bot's response text to your cloned ElevenLabs voice. Build a WAV in memory with stdlib wave, play it locally with winsound, and upload it to Discord simultaneously — all from a background thread.

// CONCEPTS

  • ElevenLabs PCM endpoint — /v1/text-to-speech/{voice_id}?output_format=pcm_22050. Returns raw signed 16-bit PCM at 22050 Hz. No ffmpeg needed.
  • stdlib wave — build a valid WAV container in memory: wave.open(buf, 'wb') → set channels/sampwidth/framerate → writeframes(pcm). No external audio lib required.
  • winsound.PlaySound — blocks until the WAV finishes playing. Use only inside a background thread or it freezes the event loop.
  • asyncio.run_coroutine_threadsafe — the only safe way to schedule a coroutine (like channel.send()) from a non-async thread. Takes the coro and the running event loop.
  • threading.Thread(daemon=True) — daemon threads die when the main process exits. Use for TTS so bot shutdown doesn't hang on audio playback.
// REFERENCE — mrrobot.py, ElevenLabs PCM fetch + speak()
import os, io, wave, threading, tempfile, asyncio
import httpx
import discord

_EL_API_KEY  = os.environ.get("EL_API_KEY", "")
_EL_VOICE_ID = os.environ.get("EL_VOICE_ID", "")
_EL_RATE     = 22050

def _fetch_el_pcm(text):
    """Call ElevenLabs, return raw PCM bytes or None on failure."""
    if not _EL_API_KEY or not _EL_VOICE_ID:
        return None
    try:
        url  = f"https://api.elevenlabs.io/v1/text-to-speech/{_EL_VOICE_ID}"
        hdrs = {"xi-api-key": _EL_API_KEY, "Content-Type": "application/json"}
        body = {
            "text": text[:5000],
            "model_id": "eleven_multilingual_v2",
            "voice_settings": {
                "stability":         0.30,
                "similarity_boost":  0.80,
                "style":             0.55,
                "use_speaker_boost": True,
            },
        }
        r = httpx.post(url, json=body, headers=hdrs,
                       params={"output_format": "pcm_22050"}, timeout=30)
        return r.content if r.status_code == 200 else None
    except Exception:
        return None

def speak(text, channel=None, loop=None):
    def _run():
        pcm = _fetch_el_pcm(text)
        if pcm is None:
            return
        buf = io.BytesIO()
        with wave.open(buf, 'wb') as wf:
            wf.setnchannels(1)
            wf.setsampwidth(2)
            wf.setframerate(_EL_RATE)
            wf.writeframes(pcm)
        tmp = tempfile.NamedTemporaryFile(
            prefix='vader_tts_', suffix='.wav', delete=False)
        tmp.write(buf.getvalue()); tmp.close()
        try:
            import winsound
            upload_fut = None
            if channel is not None and loop is not None:
                async def _upload():
                    await channel.send(
                        file=discord.File(tmp.name, filename="voice.wav"))
                upload_fut = asyncio.run_coroutine_threadsafe(_upload(), loop)
            winsound.PlaySound(tmp.name, winsound.SND_FILENAME)
            if upload_fut is not None:
                try: upload_fut.result(timeout=15)
                except Exception: pass
        finally:
            try: os.unlink(tmp.name)
            except Exception: pass
    threading.Thread(target=_run, daemon=True).start()
